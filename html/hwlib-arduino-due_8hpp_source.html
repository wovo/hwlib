509 class d2_36kHz : public hwlib::pin_out {
510  public:
511 
513  d2_36kHz(){
514 
515  // this sets the main clock to 84 MHz
516  hwlib::wait_ms( 1 );
517 
518  // enable the clock to port B
519  PMC->PMC_PCER0 = 1 << ID_PIOB;
520 
521  // disable PIO Control on PB25 and set up for Peripheral B TIOA0
522  PIOB->PIO_PDR = PIO_PB25;
523  PIOB->PIO_ABSR |= PIO_PB25;
524 
525  // enable output on B25
526  PIOB->PIO_OER = PIO_PB25;
527 
528  // enable the clock to the TC0 (peripheral # 27)
529  PMC->PMC_PCER0 = 1 << ID_TC0;
530 
531  // configure TC0 channel 0 for WAVSEL=10 mode
532  // using the main clock / 2 as source
533  // and reset on C match
534  TC0->TC_CHANNEL[ 0 ].TC_CMR =
535  TC_CMR_WAVE
536  | TC_CMR_TCCLKS_TIMER_CLOCK1
537  | TC_CMR_WAVSEL_UP_RC;
538 
539  // configure pin A match clear
540  TC0->TC_CHANNEL[ 0 ].TC_CMR |= TC_CMR_ACPC_SET | TC_CMR_ACPA_CLEAR;
541 
542  // set C match/clear for 36 kHz
543  TC0->TC_CHANNEL[ 0 ].TC_RC = 41'000'000 / 36'000;
544 
545  // set 50% duty cycle
546  TC0->TC_CHANNEL[ 0 ].TC_RA = TC0->TC_CHANNEL[ 0 ].TC_RC / 2;
547 
548  write( 0 );
549 
550  // enable and start
551  TC0->TC_CHANNEL[ 0 ].TC_CCR = TC_CCR_CLKEN | TC_CCR_SWTRG;
552  }
553 
555  //
558  void write( bool b ) override {
559  if( b ){
560  // eable C match -> pin set
561  TC0->TC_CHANNEL[ 0 ].TC_CMR |= TC_CMR_ACPC_SET;
562  } else {
563  // remove pin C match action
564  TC0->TC_CHANNEL[ 0 ].TC_CMR &= ~ ( 0x3 << 18);
565  }
566  }

741 uint_fast64_t HWLIB_WEAK now_ticks(){
742  static bool init_done = false;
743  if( ! init_done ){
744 
745  // switch to the 84 MHz crystal/PLL clock
746  sam3xa::SystemInit();
747 
748  EFC0->EEFC_FMR = EEFC_FMR_FWS(4);
749  EFC1->EEFC_FMR = EEFC_FMR_FWS(4);
750 
751  SysTick->CTRL = 0; // stop the timer
752  SysTick->LOAD = 0xFFFFFF; // use its as a 24-bit timer
753  SysTick->VAL = 0; // clear the timer
754  SysTick->CTRL = 5; // start the timer, 1:1
755 
756  init_done = true;
757  }
758 
759  static unsigned int last_low = 0;
760  static unsigned long long int high = 0;
761 
762  // the timer ticks down, but we want an up counter
763  unsigned int low = 0xFFFFFF - ( SysTick->VAL & 0xFFFFFF );
764  if( low < last_low ){
765 
766  // the timer rolled over, so increment the high part
767  high += 0x1ULL << 24;
768  }
769  last_low = low;
770 
771  // return the aggregated ticks value
772  // the counter runs at 84 MHz
773  return ( low | high );
774 
775 }
