125 GPIO_TypeDef & __attribute__((weak)) port_registers( uint32_t port ){
126 
127  // a bit of a cludge to put this here:
128  // enable the clock to all GPIO ports
129  RCC->APB2ENR |=
130  RCC_APB2ENR_IOPAEN | RCC_APB2ENR_IOPBEN |
131  RCC_APB2ENR_IOPCEN | RCC_APB2ENR_IOPDEN;
132 
133  switch( port ){
134  case 0 : return *GPIOA;
135  case 1 : return *GPIOB;
136  case 2 : return *GPIOC;
137  case 3 : return *GPIOD;
138  default : break;
139  }
140 
141  // doesn't return

145 class pin_base {
146 public:
147  volatile GPIO_TypeDef & port;
148  volatile uint32_t & config_word;
149  uint32_t pin;
150  uint32_t config_offset;
151  uint32_t mask;
152 
153  void config( uint32_t conf ){
154  config_word &= ~( 0xF << config_offset );
155  config_word |= conf << config_offset;
156  }
157 
158  pin_base( uint32_t port_number, uint32_t pin_number, uint32_t conf ):
159  port{ port_registers( port_number ) },
160  config_word{ ( pin_number < 8 ) ? port.CRL : port.CRH },
161  pin{ pin_number },
162  config_offset{ 4 * ( pin_number % 8 ) },
163  mask{ 0x1U << pin_number }
164  {
165  config( conf );
166  }
167 
168  bool base_get(){
169  return (( port.IDR & mask ) != 0 );
170  }
171 
172  void base_set( bool v ){
173  port.BSRR |= ( v ? mask : ( mask << 16 ));
174  }
175 };

336 uint_fast64_t HWLIB_WEAK now_ticks(){
337 
338  static bool init_done = false;
339  if( ! init_done ){
340 /*
341  // switch to the 72 MHz crystal/PLL clock, from stm32x.cpp,
342  // some values taken from
343  // https://github.com/rogerclarkmelbourne/STM32duino-bootloader
344 
345  // Flash 2 wait state
346  FLASH->ACR &= (uint32_t)((uint32_t)~FLASH_ACR_LATENCY);
347  FLASH->ACR |= (uint32_t)FLASH_ACR_LATENCY_2;
348 
349  // Enable Prefetch Buffer
350  FLASH->ACR |= FLASH_ACR_PRFTBE;
351 
352  // enable HSE and wait for it
353  RCC->CR |= RCC_CR_HSEON;
354  while((RCC->CR & RCC_CR_HSERDY ) == 0 ){}
355 
356 
357  // PLL configuration: PLLCLK = HSE * 9 = 72 MHz
358  RCC->CFGR &= (uint32_t)((uint32_t)~(
359  RCC_CFGR_PLLSRC | RCC_CFGR_PLLXTPRE | RCC_CFGR_PLLMULL ));
360  RCC->CFGR |= (uint32_t)( RCC_CFGR_PLLSRC | RCC_CFGR_PLLMULL9 );
361 
362  // Enable PLL and wait for it
363  RCC->CR |= RCC_CR_PLLON;
364  while((RCC->CR & RCC_CR_PLLRDY) == 0){}
365 
366  // HCLK = SYSCLK
367  RCC->CFGR |= (uint32_t)RCC_CFGR_HPRE_DIV1;
368 
369  // PCLK2 = HCLK
370  RCC->CFGR |= (uint32_t)RCC_CFGR_PPRE2_DIV1;
371 
372  // PCLK1 = HCLK / 2
373  RCC->CFGR |= (uint32_t)RCC_CFGR_PPRE1_DIV2;
374 
375  // Select PLL as system clock source
376  RCC->CFGR &= (uint32_t)((uint32_t)~(RCC_CFGR_SW));
377  RCC->CFGR |= (uint32_t)RCC_CFGR_SW_PLL;
378 
379  // Wait till PLL is used as system clock source
380  while ((RCC->CFGR & (uint32_t)RCC_CFGR_SWS) != (uint32_t)0x08){}
381 */
382  // start the systick timer
383  SysTick->CTRL = 0; // stop the timer
384  SysTick->LOAD = 0xFFFFFF; // use its as a 24-bit timer
385  SysTick->VAL = 0; // clear the timer
386  SysTick->CTRL = 5; // start the timer, 1:1
387 
388  init_done = true;
389  }
390 
391  static unsigned int last_low = 0;
392  static unsigned long long int high = 0;
393 
394  // the timer ticks down, but we want an up counter
395  unsigned int low = 0xFFFFFF - ( SysTick->VAL & 0xFFFFFF );
396  if( low < last_low ){
397 
398  // the timer rolled over, so increment the high part
399  high += 0x1ULL << 24;
400  }
401  last_low = low;
402 
403  // return the aggregated ticks value
404  // the counter runs at 84 MHz
405  return ( low | high );
406 
407 }
