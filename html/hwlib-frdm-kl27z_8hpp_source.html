190 uint_fast64_t HWLIB_WEAK now_ticks(){
191  static bool init_done = false;
192  if( ! init_done ){
193 
194  SysTick->CTRL = 0; // stop the timer
195  SysTick->LOAD = 0xFFFFFF; // use its as a 24-bit timer
196  SysTick->VAL = 0; // clear the timer
197  SysTick->CTRL = 5; // start the timer, 1:1
198 
199  init_done = true;
200  }
201 
202  static unsigned int last_low = 0;
203  static unsigned long long int high = 0;
204 
205  // the timer ticks down, but we want an up counter
206  unsigned int low = 0xFFFFFF - ( SysTick->VAL & 0xFFFFFF );
207  if( low < last_low ){
208 
209  // the timer rolled over, so increment the high part
210  high += 0x1ULL << 24;
211  }
212  last_low = low;
213 
214  // return the aggregated ticks value
215  return ( low | high );
216 
217 }

230 void uart_init(){
231  static bool init_done = false;
232  if( init_done ){
233  return;
234  }
235  init_done = true;
236 
237  // enable the clock to port A
238  PMC->PMC_PCER0 = 1 << ID_PIOA;
239 
240  // disable PIO Control on PA9 and set up for Peripheral A
241  PIOA->PIO_PDR = PIO_PA8;
242  PIOA->PIO_ABSR &= ~PIO_PA8;
243  PIOA->PIO_PDR = PIO_PA9;
244  PIOA->PIO_ABSR &= ~PIO_PA9;
245 
246  // enable the clock to the UART
247  PMC->PMC_PCER0 = ( 0x01 << ID_UART );
248 
249  // Reset and disable receiver and transmitter.
250  hw_uart->UART_CR = UART_CR_RSTRX | UART_CR_RSTTX | UART_CR_RXDIS | UART_CR_TXDIS;
251 
252  // Set the baudrate to 115200.
253  //hw_uart->UART_BRGR = 45; // MASTER_CLK_FREQ / (16 * 45) = 116666 (close enough).
254  //uart->UART_BRGR = 546; // For ~9600 baud.
255  hw_uart->UART_BRGR = 5241600 / BMPTK_BAUDRATE;
256 
257  // No parity, normal channel mode.
258  hw_uart->UART_MR = UART_MR_PAR_NO;
259 
260  // Disable all interrupts.
261  hw_uart->UART_IDR = 0xFFFFFFFF;
262 
263  // Enable the receiver and thes trasmitter.
264  hw_uart->UART_CR = UART_CR_RXEN | UART_CR_TXEN;
265 }
