554 class line : public drawable {
555 private:
556  location end;
557  color fg;
558 
559  static void swap( int_fast16_t & a, int_fast16_t & b ){
560  auto t = a;
561  a = b;
562  b = t;
563  }
564 
565  static int_fast16_t abs( int_fast16_t x ){
566  return x >= 0 ? x : -x;
567  }
568 
569 public:
571  line( location start, location end, color fg = black )
572  : drawable{ start }, end{ end }, fg{ fg }
573  {}
574 
575  void draw( window & w ) override {
576 
577  int_fast16_t x0 = start.x;
578  int_fast16_t y0 = start.y;
579  int_fast16_t x1 = end.x;
580  int_fast16_t y1 = end.y;
581 
582  // http://en.wikipedia.org/wiki/Bresenham%27s_line_algorithm
583  // http://homepages.enterprise.net/murphy/thickline/index.html
584 
585  int_fast16_t Dx = x1 - x0;
586  int_fast16_t Dy = y1 - y0;
587 
588  int_fast16_t steep = (abs(Dy) >= abs(Dx));
589 
590  if( steep ){
591  swap( x0, y0 );
592  swap( x1, y1 );
593 
594  // recompute Dx, Dy after swap
595  Dx = x1 - x0;
596  Dy = y1 - y0;
597  }
598 
599  int_fast16_t xstep = 1;
600  if( Dx < 0 ){
601  xstep = -1;
602  Dx = -Dx;
603  }
604 
605  int_fast16_t ystep = 1;
606  if( Dy < 0 ){
607  ystep = -1;
608  Dy = -Dy;
609  }
610  int_fast16_t TwoDy = 2*Dy;
611  int_fast16_t TwoDyTwoDx = TwoDy - 2*Dx; // 2*Dy - 2*Dx
612  int_fast16_t E = TwoDy - Dx; //2*Dy - Dx
613  int_fast16_t y = y0;
614  int_fast16_t xDraw, yDraw, x;
615  for( x = x0; x != x1; x += xstep ){
616  if (steep) {
617  xDraw = y;
618  yDraw = x;
619  } else {
620  xDraw = x;
621  yDraw = y;
622  }
623 
624  w.write( location( xDraw, yDraw ), fg );
625 
626  if( E > 0 ){
627  E += TwoDyTwoDx; //E += 2*Dy - 2*Dx;
628  y = y + ystep;
629  } else {
630  E += TwoDy; //E += 2*Dy;
631  }
632  }
633  }
634 
635 }; // class line

639 class circle : public drawable {
640 private:
641  uint_fast16_t radius;
642  color fg;
643  color bg;
644 
645 public:
647  circle( location start, uint_fast16_t radius, color fg = black, color bg = transparent )
648  : drawable{ start }, radius{ radius }, fg{ fg }, bg{ bg }
649  {}
650 
651  void draw( window & w ) override {
652 
653  // don't draw anything when the size would be 0
654  if( radius < 1 ){
655  return;
656  }
657 
658  // http://en.wikipedia.org/wiki/Midpoint_circle_algorithm
659 
660  int_fast16_t fx = 1 - radius;
661  int_fast16_t ddFx = 1;
662  int_fast16_t ddFy = -2 * radius;
663  int_fast16_t x = 0;
664  int_fast16_t y = radius;
665 
666  // top and bottom
667  w.write( start + location( 0, + radius ), fg );
668  w.write( start + location( 0, - radius ), fg );
669 
670  // left and right
671  w.write( start + location( + radius, 0 ), fg );
672  w.write( start + location( - radius, 0 ), fg );
673 
674  if( bg != transparent ){
675 
676  // top and bottom
677  w.write( start + location( 0, + radius ), fg );
678  w.write( start + location( 0, - radius ), fg );
679 
680  // left and right
681  line(
682  start - location( radius, 0 ),
683  start + location( radius, 0 ),
684  fg
685  ).draw( w );
686  }
687 
688  while( x < y ){
689 
690  // calculate next outer circle point
691  if( fx >= 0 ){
692  y--;
693  ddFy += 2;
694  fx += ddFy;
695  }
696  x++;
697  ddFx += 2;
698  fx += ddFx;
699 
700  w.write( start + location( + x, + y ), fg );
701  w.write( start + location( - x, + y ), fg );
702  w.write( start + location( + x, - y ), fg );
703  w.write( start + location( - x, - y ), fg );
704  w.write( start + location( + y, + x ), fg );
705  w.write( start + location( - y, + x ), fg );
706  w.write( start + location( + y, - x ), fg );
707  w.write( start + location( - y, - x ), fg );
708 
709  if( bg != transparent ){
710  line(
711  start + location( -x, y ),
712  start + location( x, y ),
713  bg ).draw( w );
714  line(
715  start + location( -x, -y ),
716  start + location( x, -y ),
717  bg ).draw( w );
718  line(
719  start + location( -y, x ),
720  start + location( y, x ),
721  bg ).draw( w );
722  line(
723  start + location( -y, -x ),
724  start + location( y, -x ),
725  bg ).draw( w );
726  }
727  }
728  }
