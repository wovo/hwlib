122  // configure a pin as ADC
123  uint32_t HWLIB_WEAK configure_as_adc( uint32_t port, uint32_t pin ){
124 
125  // enable IOCON & A/D clock
126  LPC_SYSCON->SYSAHBCLKCTRL |= ( 0x01 << 16 ) | ( 0x01 << 13 );
127 
128  // enable A/D power
129  LPC_SYSCON->PDRUNCFG &= ~( 1 << 4 );
130 
131  switch( ( port << 8 ) | pin ){
132  case 0x000B: pin_ad_conf( &LPC_IOCON->R_PIO0_11, 0x02 ); return 0;
133 
134  case 0x0100: pin_ad_conf( &LPC_IOCON->R_PIO1_0, 0x02 ); return 1;
135  case 0x0101: pin_ad_conf( &LPC_IOCON->R_PIO1_1, 0x02 ); return 2;
136  case 0x0102: pin_ad_conf( &LPC_IOCON->R_PIO1_2, 0x02 ); return 3;
137  case 0x0103: pin_ad_conf( &LPC_IOCON->SWDIO_PIO1_3, 0x02 ); return 4;
138  case 0x0104: pin_ad_conf( &LPC_IOCON->PIO1_4, 0x01 ); return 5;
139 
140  default: HWLIB_PANIC_WITH_LOCATION;

302 class pin_adc : public hwlib::adc {
303 private:
304  uint32_t channel;
305 
306 public:
307 
309  pin_adc( uint32_t port, uint32_t pin )
310  : adc{ 10 }, channel{ configure_as_adc( port, pin ) }
311  {}
312 
313  adc_value_type read() override {
314 
315  // configure the A/D for the pin
316  LPC_ADC->CR = ( 0x01 << channel ) | ( 12 << 8 );
317 
318  // start the conversion
319  LPC_ADC->CR |= ( 0x01 << 24 );
320 
321  // wait for the conversion to complete
322  while( ( LPC_ADC->GDR & ( 1 << 31 )) == 0 );
323 
324  hwlib::wait_ms( 10 );
325 
326  // return the A/D result
327  return 3 | (( LPC_ADC->GDR >> 6 ) & 0x3FF);
328  }
329 
330  void refresh() override {
331  }

345 void HWLIB_WEAK wait_us( int_fast32_t n ){
346  // for the default clock frequency of 12 MHz:
347  // the loop uses 12 cycles
348  __asm volatile(
349  " mov r0, %[reg] \t\n"
350  " b 1f \t\n"
351  " .align 4 \t\n"
352  "1: sub r0, #1 \t\n"
353  " b 2f \t\n"
354  "2: b 3f \t\n"
355  "3: nop \t\n"
356  " nop \t\n"
357  " bgt 1b \t\n"
358  :: [reg] "r" (n) : "r0"
359  );
360 }
361 
362 void HWLIB_WEAK wait_ms( int_fast32_t n ){
363  while( n > 0 ){
364  wait_us( 1000 );
365  --n;
366  }
